IRS_mnt_notify

#include <irs.h>

typedef struct IRS_mnt_notify {  //
    struct IRS_header  IRS_m_n_hdr;
    USHORT IRS_m_n_drive;            // Supplied Drive number (0 = A:, ...)
    void   *IRS_m_n_pvrp;            // Supplied VRP pointer
    ULONG  IRS_m_n_drivemap;         // Returned Bitmap of child volumes
    ULONG  IRS_m_n_effective_drive;  // Returned Effective drive number
    ULONG  IRS_m_n_actual_drive;     // Returned Actual drive number
} IRS_mnt_notify, *PIRS_mnt_notify;

Contains information for a IRS_MOUNT_NOTIFY function.

IRS_m_n_hdr.IRS_result
Result. Can be IRS_M_N_SUCCESS to indicate nothing to do or IRS_M_N_REMOUNT to remount the drive.
