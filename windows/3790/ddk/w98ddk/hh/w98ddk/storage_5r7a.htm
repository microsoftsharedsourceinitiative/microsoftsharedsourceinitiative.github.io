AEP_bi_init

#include <aep.h>

typedef struct AEP_bi_init { 
    struct AEPHDR  AEP_bi_i_hdr;   // Standard Header
    ULONG  AEP_bi_reference_data;  // optional value passed in from DRP 
    UCHAR  AEP_bi_flags;           // flags; see below
    CHAR   AEP_bi_i_max_target;    // max scsi target id
    CHAR   AEP_bi_i_max_lun;       // max scsi lun
    ULONG  AEP_bi_i_dcb;           // init DCB for port drivers
    PVOID  AEP_bi_i_hdevnode;      // handle to devnode for this adapter
    PVOID  AEP_bi_i_regkey;        // registry key for this adapter
    UCHAR  AEP_bi_i_align[1];      // reserved; don't use
} AEP_bi_init, *PAEP_bi_init;

Contains information for an AEP_INITIALIZE function.

AEP_bi_flags
Flags. Can be one of these values:
AEP_BI_FL_SCSI_SCAN_DOWN (0x01) Device requires reverse ID scanning.
AEP_BI_FL_SEND_CONFIG_AGAIN (0x02) Device wants this configuration information again. If this is set, IOS reexecutes the entire sequence of AEP_INITIALIZE, AEP_INQUIRE, and AEP_CONFIG. This is typically not needed since IOS calls the port driver again if the port driver controls more than one controller.
