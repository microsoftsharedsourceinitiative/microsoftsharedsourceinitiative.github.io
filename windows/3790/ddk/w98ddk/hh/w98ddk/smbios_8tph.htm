SMBIOS Example

The schema for the raw SMBIOS information is reproduced in the following code extract. 

// SMBIOS data classes
//
[abstract]
class WMISMmBios
{
};
 
[Dynamic, Provider("WMIProv"),
 WMI,
 Description("Raw SMBIOS Tables"),
 guid("{8F680850-A584-11d1-BF38-00A0C9062910}"),
 locale("MS\\0x409")
 
]
class RawSMBiosTables : WMISmBios
{
    [key, read]
     string InstanceName;
    [read] boolean Active;
 
    [WmiDataId(1),
     read
    ] uint32 Size;
 
    [WmiDataId(2),
     WmiSizeIs("Size"),
     read
    ] uint8 SMBiosData[];
 
};
