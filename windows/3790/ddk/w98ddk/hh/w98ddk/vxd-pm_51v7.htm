Backward Compatibility of Power Management Messages

For compatibility with Windows 98 drivers, some of the preceding messages are defined to be the same as older messages.

#define CONFIG_SET_POWERSTATE_D1                CONFIG_APM_STANDBY
#define CONFIG_SET_POWERSTATE_D2                CONFIG_APM_SUSPEND
#define CONFIG_TEST_SET_POWERSTATE_D1           CONFIG_APM_TEST_STANDBY
#define CONFIG_TEST_SET_POWERSTATE_D2           CONFIG_APM_TEST_SUSPEND
#define CONFIG_TEST_SET_POWERSTATE_D1_FAILED    CONFIG_APM_TEST_STANDBY_FAILED
#define CONFIG_TEST_SET_POWERSTATE_D2_FAILED    CONFIG_APM_TEST_SUSPEND_FAILED
