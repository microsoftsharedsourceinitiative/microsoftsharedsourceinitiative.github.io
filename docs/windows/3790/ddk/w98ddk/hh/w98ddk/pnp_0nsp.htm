MEM_RANGE

struct Mem_Range_s {
    ULONG MR_Align;     // mask for base alignment
    ULONG MR_nBytes;    // count of bytes
    ULONG MR_Min;       // min Address
    ULONG MR_Max;       // max Address
    WORD  MR_Flags;     // flags
    WORD  MR_Reserved;  // reserved; do not use
};
typedef struct Mem_Range_s MEM_RANGE;
